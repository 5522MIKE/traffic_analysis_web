## 目录结构
```
├── src/ # 源文件目录
	├── api/
        ├── api.js # 构建Axios实例
        ├── global.js # 全局变量
        ├── index.js # 包的引入
    ├── assets/ # 静态文件
    ├── components/ # 组件
        ├── backGround.vue # 背景组件
        ├── HelloWorld.vue # 测试用组件
        ├── information.vue # 违规信息展示组件
        ├── Main.vue # 主组件
        ├── number.vue # 机动车及行人数量可视化组件
        ├── vehicle.vue # 车辆违规数据可视化组件
        ├── VideoPlay.vue # 视频播放组件组件
    ├── router/
        ├── index.js # 路由
    ├── App.vue
    ├── main.js
├── UML/ # UML文件
	├──薪酬系统-实践.asta  # 范例建模
├── package.json
├── README.md
├── README_vue.md
```
